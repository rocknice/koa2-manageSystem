 * @class

const rp = require('request-promise')

class Book {
    /**
     * @constructor
     * @param {string} app KOA2执行上下文
     */
    constructor(app){}
    /**
     * 获取后台全部图书的数据方法
     * @params {*} options 配置项
     * @example
     * return new Promise
     * actionInit()
     */
    actionInit() {
        const options = {
            uri: 'http://localhost:8080/index.php?r=books%2Findex',
            method: "GET",
            json: true,
            // headers: {
            //     "Content-Type": "application/json; charset=utf-8"
            // }
        };
        return new Promise((resolve, reject) => {
            rp(options)
            .then(function (res) {
                if (res) {
                    resolve(res)
                }
                resolve(res.data)
            })
            .catch(function (err) {
                reject(err)
            });
        })
    }
    actionView(params) {
